//İstenilen dizi
  let arr = [22, 27, 16, 2, 18, 6];

  /**
   * Algoritmanın işlendiği fonksiyon.
   * @param  {Array} params
   */
  function insertionSort(params) {
    //Dizi uzunluğu: n
    let n = params.length;

    // Diziyi baştan sona gezecek olan döngü.
    for (let i = 0; i < n; i++) {
      /**
       * tutucu: Takas edilecek olan veri.
       * imlec: En küçük değere sahip olan verinin göstergesi.
       **/
      let tutucu = params[i], imlec = 0;
      // En son yerleştirilen elemandan itibaren gezecek olan döngü.
      for (let j = i; j < n; j++) {
        // Takas saptaması
        if (tutucu > params[j]) {
          imlec = j;
          tutucu = params[j];
        }
      }
      // Takasın sağlanması
      if (imlec != 0) {
        params[imlec] = params[i];
        params[i] = tutucu;
        document.write(JSON.stringify(params) + "\n");
      }
    }
    return params;
  }

  insertionSort(arr);
